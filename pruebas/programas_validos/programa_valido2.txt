int[3][3] matrizA;
int[3][3] matrizB;
int[3][3] resultado;

def int main() {
    int i;
    int j;
    
    for (i = 0; i < 3; i = i + 1) {
        for (j = 0; j < 3; j = j + 1) {
            matrizA[i][j] = i + j;
            matrizB[i][j] = i * j;
        }
    }

    i = 0;
    while (i < 3) {
        j = 0;
        while (j < 3) {
            resultado[i][j] = matrizA[i][j] + matrizB[i][j];
            j = j + 1;
        }
        i = i + 1;
    }
    
    return 0;
}
